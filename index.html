from flask import Flask, jsonify, send_file from flask_cors import CORS import requests import os app = Flask(__name__) CORS(app) SLEEPER = "https://api.sleeper.app/v1" # Serve the frontend @app.route("/") def index(): return send_file("index.html") @app.route("/user/") def get_user(username): r = requests.get(f"{SLEEPER}/user/{username}") return jsonify(r.json()) @app.route("/leagues/") def get_leagues(user_id): r = requests.get(f"{SLEEPER}/user/{user_id}/leagues/nfl/2025") return jsonify(r.json()) @app.route("/league//users") def get_users(league_id): r = requests.get(f"{SLEEPER}/league/{league_id}/users") return jsonify(r.json()) @app.route("/league//rosters") def get_rosters(league_id): r = requests.get(f"{SLEEPER}/league/{league_id}/rosters") return jsonify(r.json()) @app.route("/league//info") def get_info(league_id): r = requests.get(f"{SLEEPER}/league/{league_id}") return jsonify(r.json()) @app.route("/league//matchups/") def get_matchups(league_id, week): r = requests.get(f"{SLEEPER}/league/{league_id}/matchups/{week}") return jsonify(r.json()) if __name__ == "__main__": port = int(os.environ.get("PORT", 8080)) app.run(host="0.0.0.0", port=port)
